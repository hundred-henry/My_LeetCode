class Solution {
public:
    int largestAltitude(vector<int>& gain) {
        int max_alt=gain[0];
        if (max_alt < 0){
            max_alt = 0;
        }
        for (int i=1; i<gain.size(); i++){
            gain[i] += gain[i-1];
            if (max_alt<gain[i]){
                max_alt = gain[i];
            }
        }
        return max_alt;
    }
};
